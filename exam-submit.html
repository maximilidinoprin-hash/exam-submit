function doPost(e) { const data = JSON.parse(e.postData.contents); const folderId = "YOUR_FOLDER_ID"; // ← ใส่ Folder ID ของคุณ const folder = DriveApp.getFolderById(folderId); // สร้างโฟลเดอร์ตามวิชา const subFolder = getOrCreate(folder, data.course); const examFolder = getOrCreate(subFolder, data.examType); // บันทึกแต่ละไฟล์ data.files.forEach(f => { const bytes = Utilities.base64Decode(f.data); const blob = Utilities.newBlob(bytes, f.type, f.name); examFolder.createFile(blob); }); // บันทึก log ลง Sheet const ss = SpreadsheetApp.openById("YOUR_SHEET_ID"); ss.getActiveSheet().appendRow([ new Date(), data.studentId, data.name, data.section, data.seat, data.course, data.examType, data.files.map(f=>f.name).join(", ") ]); return ContentService .createTextOutput(JSON.stringify({ok:true})) .setMimeType(ContentService.MimeType.JSON); } function getOrCreate(parent, name) { const found = parent.getFoldersByName(name); return found.hasNext() ? found.next() : parent.createFolder(name); }
